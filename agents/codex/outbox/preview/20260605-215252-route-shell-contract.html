Route Metadata Shape

import type { Href, Route } from "expo-router";
import type { NativeStackNavigationOptions } from "@react-navigation/native-stack";

export type Locale = "en" | "ko";

export type AuthRequirement =
  | "public"             // no auth requirement
  | "guestOnly"          // sign-in/sign-up; signed-in users redirect to /
  | "signedIn"           // any signed-in user; profile may still be incomplete
  | "profileComplete"    // signed in + public.users profile/consent gate complete
  | "profileIncomplete"; // only the post-OAuth/profile-completion lane

export type ShellVisibility =
  | "app"       // app chrome may render; route participates in BackArrow/tab/spacing
  | "auth"      // auth surface; no app nav chrome
  | "redirect"  // alias route that should immediately redirect/replace
  | "bare";     // framework/not-found escape hatch

export type BackOffsetPolicy = "default" | "afterTabBrand";

export type BackBehavior =
  | { kind: "hidden" }
  | { kind: "home"; action: "push" | "replace"; offset?: BackOffsetPolicy }
  | { kind: "route"; action: "push" | "replace"; href: Href; offset?: BackOffsetPolicy }
  | { kind: "native"; fallback: Href; offset?: BackOffsetPolicy };

export interface LocalizedLabel {
  en: string;
  ko: string;
}

export interface RouteLabelContext {
  locale: Locale;
  pathname: string;
  params: Record<string, string | string[] | undefined>;
}

export type RouteLabel = LocalizedLabel | ((ctx: RouteLabelContext) => string | null);

export type TabId = "graph" | "capture" | "jarvis" | "profile";
export type TabIconId = "graph" | "capture" | "jarvis" | "profile";

export interface TabMeta {
  id: TabId;
  order: number;
  icon: TabIconId;
  href: Href;
  label: LocalizedLabel;
  navigation: "replace";
}

export interface ReplacementRoute {
  href: Href;
  action: "redirect" | "replace";
  preserveParams?: true | readonly string[];
  note?: string;
}

export interface RouteMeta {
  id: string;
  path: string;              // URL pattern, e.g. "/", "/record/[id]"
  match?: RegExp;            // required for dynamic URLs
  stackName?: string;        // Expo Stack.Screen name, e.g. "index", "record/[id]", "(auth)"
  stackOptions?: NativeStackNavigationOptions;
  label: RouteLabel;
  shell: ShellVisibility;
  tab: false | TabMeta;
  auth: AuthRequirement;
  back: BackBehavior;
  replacement?: ReplacementRoute;
}

Required Route APIs

export const ROUTES = [...] as const satisfies readonly RouteMeta[];

export function getRouteMeta(pathname: string): RouteMeta {
  return ROUTES.find((route) =>
    route.match ? route.match.test(pathname) : route.path === pathname
  ) ?? NOT_FOUND_ROUTE;
}

export function getRouteLabel(pathname: string, locale: Locale, params = {}): string | null {
  const meta = getRouteMeta(pathname);
  return typeof meta.label === "function"
    ? meta.label({ pathname, locale, params })
    : meta.label[locale];
}

export function getTabRoutes(): TabMeta[] {
  return ROUTES
    .filter((route): route is RouteMeta & { tab: TabMeta } => route.tab !== false)
    .map((route) => route.tab)
    .sort((a, b) => a.order - b.order);
}

export function shouldShowTabBar(pathname: string): boolean {
  return getRouteMeta(pathname).tab !== false;
}

export function getBackBehavior(pathname: string): BackBehavior {
  const meta = getRouteMeta(pathname);
  return meta.back.kind === "hidden" ? meta.back : {
    ...meta.back,
    offset: meta.tab ? "afterTabBrand" : meta.back.offset ?? "default",
  };
}

export function canonicalHref(href: Href): Href {
  const pathname = typeof href === "string" ? href : String(href.pathname);
  const replacement = getRouteMeta(pathname).replacement;
  if (!replacement) return href;
  if (replacement.preserveParams === true && typeof href !== "string") {
    return { pathname: String((replacement.href as { pathname?: string }).pathname ?? replacement.href), params: href.params };
  }
  return replacement.href;
}

Shell Spacing Shape

import type { EdgeInsets } from "react-native-safe-area-context";
import { spacing } from "@/lib/theme/tokens";
import { getBackBehavior, getRouteMeta, shouldShowTabBar } from "@/lib/nav/routes";

export const SHELL_SPACING = {
  screen: {
    horizontal: spacing.lg,
    topBase: spacing.sm,
    scrollBottomBase: spacing.xl,
  },
  backArrow: {
    size: 44,
    topGap: spacing.sm,       // current insets.top + 8
    leftGap: spacing.md,      // current insets.left + 12
    tabBrandOffset: 52,       // current primary-tab nudge
    labelGap: spacing.sm,
    labelMaxWidth: 220,
    reservedHeadroom: 60,     // replaces PremiumAppShell arrowHeadroom literal
  },
  tabBar: {
    contentHeight: 62,        // current TAB_BAR_HEIGHT
    horizontalPadding: spacing.sm,
    rowGap: spacing.xs,
    contentTopClearance: spacing.lg,
  },
  navGraph: {
    sheetGapAboveTabBar: 30,  // current 92 = 62 + 30 on web/zero safe bottom
    zoomOverlayPaddingBottom: "34%",
  },
} as const;

export interface ResolvedShellSpacing {
  backArrow: {
    visible: boolean;
    top: number;
    left: number;
    size: number;
    labelGap: number;
    labelMaxWidth: number;
    reserveTop: number;
  };
  tabBar: {
    visible: boolean;
    contentHeight: number;
    safeBottom: number;
    totalHeight: number;
    contentBottomClearance: number;
  };
  content: {
    paddingHorizontal: number;
    paddingTop: number;
    paddingBottom: number;
    scrollPaddingBottom: number;
  };
  navGraph: {
    sheetBottom: number;
    zoomOverlayPaddingBottom: string;
  };
}

export function resolveShellSpacing(pathname: string, insets: EdgeInsets): ResolvedShellSpacing {
  const meta = getRouteMeta(pathname);
  const tabVisible = shouldShowTabBar(pathname);
  const back = getBackBehavior(pathname);
  const backVisible = meta.shell === "app" && back.kind !== "hidden";
  const tabTotalHeight = SHELL_SPACING.tabBar.contentHeight + insets.bottom;
  const contentBottomClearance = tabVisible
    ? SHELL_SPACING.tabBar.contentHeight + SHELL_SPACING.tabBar.contentTopClearance
    : 0;

  return {
    backArrow: {
      visible: backVisible,
      top: insets.top + SHELL_SPACING.backArrow.topGap,
      left:
        insets.left +
        SHELL_SPACING.backArrow.leftGap +
        (back.offset === "afterTabBrand" ? SHELL_SPACING.backArrow.tabBrandOffset : 0),
      size: SHELL_SPACING.backArrow.size,
      labelGap: SHELL_SPACING.backArrow.labelGap,
      labelMaxWidth: SHELL_SPACING.backArrow.labelMaxWidth,
      reserveTop: backVisible ? SHELL_SPACING.backArrow.reservedHeadroom : 0,
    },
    tabBar: {
      visible: meta.shell === "app" && tabVisible,
      contentHeight: SHELL_SPACING.tabBar.contentHeight,
      safeBottom: insets.bottom,
      totalHeight: tabTotalHeight,
      contentBottomClearance,
    },
    content: {
      paddingHorizontal: SHELL_SPACING.screen.horizontal,
      paddingTop: SHELL_SPACING.screen.topBase + (backVisible ? SHELL_SPACING.backArrow.reservedHeadroom : 0),
      paddingBottom: contentBottomClearance,
      scrollPaddingBottom: SHELL_SPACING.screen.scrollBottomBase,
    },
    navGraph: {
      sheetBottom: tabVisible
        ? tabTotalHeight + SHELL_SPACING.navGraph.sheetGapAboveTabBar
        : insets.bottom + SHELL_SPACING.navGraph.sheetGapAboveTabBar,
      zoomOverlayPaddingBottom: SHELL_SPACING.navGraph.zoomOverlayPaddingBottom,
    },
  };
}

Rule: route screens should stop adding overlay clearance locally. A screen may keep ordinary content rhythm (scrollPaddingBottom) but tab/back/NavGraph overlay spacing comes from resolveShellSpacing.

Consumer Contract

  • BackArrow: replace local PRE_AUTH_PATHS, HIDDEN_PATHS, ROUTE_LABELS, titleForRoute, and tab offset math with getBackBehavior, getRouteLabel, and resolveShellSpacing.
  • PremiumTabBar: delete local TABS and TAB_PATHS; render getTabRoutes(); use spacing resolver for height and safe-bottom padding; keep icon rendering local by TabIconId.
  • Stack: generate <Stack.Screen> rows from ROUTES.filter(r => r.stackName); this brings missing files (data, theme, support, onboarding, imagine, record/[id]) under the same inventory.
  • PremiumAppShell: stop importing BackArrow helpers; call resolveShellSpacing(pathname, insets) once and apply content.paddingTop, content.paddingBottom, and horizontal padding.
  • NavGraph: canonicalize hrefs before navigation; replace bottom: 92 and paddingBottom: "34%" with resolveShellSpacing("/", insets).navGraph.
  • Route-local buttons/links: replace hard-coded /journal action targets with canonicalHref("/journal") first, then migrate copy to canonical /capture route names in a later content pass.
